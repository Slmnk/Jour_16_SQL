CREATE TABLE `users` (
`id` INTEGER PRIMARY KEY AUTOINCREMENT,
`name` TEXT,
);

CREATE TABLE `pins` (
`id` INTEGER PRIMARY KEY AUTOINCREMENT,
`url` TEXT,
`user_id` INTEGER
FOREIGN KEY(user_id) REFERENCES users(id)
);

CREATE TABLE `comments` (
`id` INTEGER PRIMARY KEY AUTOINCREMENT,
`content` TEXT,
`user_id` INTEGER,
`pin_id` INTEGER,
FOREIGN KEY(user_id) REFERENCES users(id),
FOREIGN KEY(pin_id) REFERENCES pins(id)
);
